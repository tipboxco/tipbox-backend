# Developer Ortamı Environment Variables Örneği
# Bu dosyayı .env olarak kopyalayın ve değerleri doldurun
# NOT: Docker Container içinde çalışırken servis isimlerini kullanın (postgres, redis, minio)

# Server Configuration
PORT=3000
NODE_ENV=development

# Auth0 entegrasyonu için
AUTH0_DOMAIN=your-auth0-domain.auth0.com
AUTH0_AUDIENCE=https://tipbox-backend

# Database Configuration (Docker Container - postgres servisi)
# Container içinde: postgres:5432
# Lokal erişim için: localhost:5432
DATABASE_URL=postgresql://postgres:postgres@postgres:5432/tipbox_dev

# CORS Configuration (virgülle ayrılmış)
# Developer ortamı için localhost ve tüm portlar
CORS_ORIGINS=http://localhost:3000,http://localhost:3001,http://localhost:5173
CORS_METHODS=GET,POST,PUT,DELETE,OPTIONS,PATCH

# Redis yapılandırması (Docker Container - redis servisi)
# Container içinde: redis:6379
# Lokal erişim için: localhost:6379
REDIS_URL=redis://redis:6379

# MinIO/S3 yapılandırması (Docker Container - minio servisi)
# Container içinde: minio:9000
# Lokal erişim için: localhost:9000
MINIO_ROOT_USER=minioadmin
MINIO_ROOT_PASSWORD=minioadmin123
S3_ENDPOINT=http://minio:9000
S3_BUCKET_NAME=tipbox-media
S3_REGION=eu-central-1
S3_ACCESS_KEY=minioadmin
S3_SECRET_KEY=minioadmin123

# Email (Google Workspace OAuth 2.0) yapılandırması
# Google Cloud Service Account JSON dosyasının path'i (container içinde absolute path)
GOOGLE_APPLICATION_CREDENTIALS=/app/tipboxbackend-3e2c3d3c0b31.json
# Service Account'un hangi email adına gönderim yapacağı (Domain-wide delegation ile)
EMAIL_USER_TO_IMPERSONATE=info@tipbox.co
# E-postada görünecek "Kimden" adı
EMAIL_FROM_NAME=Tipbox

# JWT Secret (Development için - production'da güçlü bir secret kullanın)
JWT_SECRET=your-super-secret-jwt-key-for-development-only
