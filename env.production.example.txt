# Production Ortamı Environment Variables Örneği
# Bu dosyayı .env.production olarak kopyalayın ve değerleri doldurun
# NOT: Docker Container içinde çalışırken servis isimlerini kullanın (postgres, redis, minio)

# Server Configuration
PORT=3000
NODE_ENV=production

# Auth0 entegrasyonu için
AUTH0_DOMAIN=your-auth0-domain.auth0.com
AUTH0_AUDIENCE=https://tipbox-backend

# Database Configuration (Docker Container - postgres servisi)
# docker-compose.prod.yml'de POSTGRES_PASSWORD env'den alınır
# Container içinde: postgres:5432
DATABASE_URL=postgresql://tipbox_user:${POSTGRES_PASSWORD}@postgres:5432/tipbox_prod

# CORS Configuration (virgülle ayrılmış)
# Production ortamı için production domain'ler
CORS_ORIGINS=https://api.tipbox.co,https://api.tipbox.co/v1,https://api.tipbox.co/v1/docs,https://app.tipbox.co
CORS_METHODS=GET,POST,PUT,DELETE,OPTIONS,PATCH

# Redis yapılandırması (Docker Container - redis servisi, password ile)
# docker-compose.prod.yml'de REDIS_PASSWORD env'den alınır
# Container içinde: redis:6379
REDIS_URL=redis://:${REDIS_PASSWORD}@redis:6379

# MinIO/S3 yapılandırması (Docker Container - minio servisi)
# docker-compose.prod.yml'de MINIO_ROOT_USER ve MINIO_ROOT_PASSWORD env'den alınır
# Container içinde: minio:9000
MINIO_ROOT_USER=${MINIO_ROOT_USER}
MINIO_ROOT_PASSWORD=${MINIO_ROOT_PASSWORD}
S3_ENDPOINT=http://minio:9000
S3_BUCKET_NAME=${S3_BUCKET_NAME}
S3_REGION=${S3_REGION}
S3_ACCESS_KEY=${MINIO_ROOT_USER}
S3_SECRET_KEY=${MINIO_ROOT_PASSWORD}

# Email (Google Workspace OAuth 2.0) yapılandırması
# Google Cloud Service Account JSON dosyasının path'i (container içinde absolute path)
GOOGLE_APPLICATION_CREDENTIALS=/app/tipboxbackend-3e2c3d3c0b31.json
# Service Account'un hangi email adına gönderim yapacağı (Domain-wide delegation ile)
EMAIL_USER_TO_IMPERSONATE=${EMAIL_USER_TO_IMPERSONATE}
# E-postada görünecek "Kimden" adı
EMAIL_FROM_NAME=${EMAIL_FROM_NAME}

# JWT Secret (Production için - güçlü bir secret kullanın!)
JWT_SECRET=${JWT_SECRET}
